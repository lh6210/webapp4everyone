Calculus 2 -- Double Integral

problem 1: evaluate \( \int_{1}^{\infty} \!\!\! \int_{x^2}^{\infty} \! \frac{1}{x^4+y^2} \, \mathrm{d}y \, \mathrm{d}x \)

\begin{equation*} \label{eq1} \begin{split} Area & = \int_{1}^{\infty} \!\!\! \int_{x^2}^{\infty} \! \frac{1}{x^4+y^2} \, \mathrm{d}y \, \mathrm{d}x \\ & = \int_{1}^{\infty}(\frac{1}{x^4} \cdot \int_{x^2}^{\infty} \frac{1}{1+(\frac{y}{x^2})^2} \, \mathrm{d}y) \, \mathrm{d}x \\ &= \int_{1}^{\infty}(\frac{1}{x^2} \cdot \int_{x^2}^{\infty} \frac{1}{1 + (\frac{y}{x^2})^2} \, \mathrm{d}(\frac{y}{x^2}) \, \mathrm{d}x \\ &= \int_{1}^{\infty}\frac{1}{x^2} \cdot arctan(\frac{y}{x^2})\Big \rvert _{x^2}^{\infty} \, \mathrm{d}x \\ &= \int_{1}^{\infty} \frac{1}{x^2} \cdot (arctan(\infty) - arctan(1)) \, \mathrm{d}x \\ &= \frac{\pi}{4} \cdot \int_{1}^{\infty} \frac{1}{x^2} \, \mathrm{d}x \\ &= \dots \end{split} \end{equation*}
